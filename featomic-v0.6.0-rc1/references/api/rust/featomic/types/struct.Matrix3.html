pub struct Matrix3(/* private fields */);
Expand description

A 3x3 square matrix type.

Matrix3 implements all the usual arithmetic operations:

use featomic::types::{Matrix3, Vector3D};

let one = Matrix3::one();
let a = Matrix3::new([
    [1.0, 0.0, 3.0],
    [0.0, 2.0, 5.6],
    [0.0, 0.0, 8.0]
]);

let v = Vector3D::new(3.0, 2.0, 1.0);

// Indexing
assert_eq!(a[0][0], 1.0);
assert_eq!(a[1][2], 5.6);

// Addition
let c = a + one;
assert_eq!(c, Matrix3::new([
    [2.0, 0.0, 3.0],
    [0.0, 3.0, 5.6],
    [0.0, 0.0, 9.0]
]));

// Subtraction
let c = a - one;
assert_eq!(c, Matrix3::new([
    [0.0, 0.0, 3.0],
    [0.0, 1.0, 5.6],
    [0.0, 0.0, 7.0]
]));

// Multiplication
let c = a * one;  // matrix - matrix
assert_eq!(c, a);

let c = a * v;  // matrix - vector
assert_eq!(c, Vector3D::new(6.0, 9.6, 8.0));

let c = 42.0 * one;  // matrix - scalar
assert_eq!(c, Matrix3::new([
    [42., 0.0, 0.0],
    [0.0, 42., 0.0],
    [0.0, 0.0, 42.]
]));

// Division
let c = a / 2.0;
assert_eq!(c, Matrix3::new([
    [0.5, 0.0, 1.5],
    [0.0, 1.0, 2.8],
    [0.0, 0.0, 4.0]
]));
